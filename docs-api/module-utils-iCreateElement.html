(inner) iCreateElement(props) → {void}

Source:
  • utils/iCreateElement.js, line 3
For simplify creating custom HTMLElements in DOM
Examples
// void: <body><div class="defaultClass"><i>my custom HTMLElement is ...</i></div></body>
iCreateElement({})
// void: <body><h3 class="defaultClass">hello h3</h3></body>
iCreateElement({ tag: 'h3', inner: 'hello h3' })
// void: <body><br></body>
iCreateElement({ tag: 'br', only: true })
// void: <body><h3></h3></body>
iCreateElement({ tag: 'h3', only: true })
// void: <body><h3 class="defaultClass"><i>my custom HTMLElement is ...</i></h3></body>
iCreateElement({ tag: 'h3', only: false })
// void: <div> * </div>
  iCreateElement({ only: 'true', inner: ' * ' })
// void: <button> * </button>
  iCreateElement({ tag: 'button', only: 'true', inner: ' * ' })
// void: <body><div class="defaultClass"><span class="i-class"><p>any html code</p></span></div></body>
iCreateElement({
      inner: '<span class="i-class"><p>any html code</p></span>',
   })
// void: <body><div class="defaultClass">html class</div></body>
iCreateElement({ tag: 'div', inner: 'html class' })
// void: <body><div class="my-best-html-class">html class</div></body>
iCreateElement({ tag: 'div', inner: 'html class', className: 'my-best-html-class' })
// void: '<button class="myButton">just a button<strong class="defaultClass">very strong</strong></button>'
iCreateElement({
        tag: 'button',
        className: 'myButton',
        inner: 'just a button',
   })
    iCreateElement({
        tag: 'strong',
        inner: 'very strong',
        parentId: '.myButton',
    })    
// void: <div id="my-best-id" class="defaultClass">html id</div>
       iCreateElement({
            tag: 'div',
            inner: 'html id',
            id: 'my-best-id    ',
        })
// void: <button id="root" class="cls">rooot<button class="defaultClass">hello</button></button>
        iCreateElement({
            id: 'root',
            className: 'cls',
            inner: 'rooot',
            tag: 'button'
        })
        iCreateElement({
            tag: 'button',
            inner: 'hello',
            parentId: '#root'
        })
// void: <p id="foo">hello</p>
  iCreateElement({id: 'foo', cut: 'all', tag: 'p', inner: 'hello'})
// void: <div id="root"></div>
  iCreateElement({ id: 'root', cut: 'all' })
Parameters:
Name Type Description
props object configure your future DOM element
Properties
Name Type Description
tag string any html tag
only boolean false by default, set true if needed to create only hollow tag, without default innerHTML and default html class
inner string any html; will append to {props.tag} you set before; have default value
className string html class for {props.tag} element, by default each {props.tag} set with html class 'defaultClass'
id string html id for {props.tag} element, by default each {props.tag} does not id
parentId string html id or html class of parent element, for appending created {props.tag} element to the end of the parent, by default parent is DOM element 'body'
cut string default / class / all - for cutting some default behavior
Returns:
creating custom DOMElement
Type
void
